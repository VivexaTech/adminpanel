rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSuperAdmin() {
      return request.auth != null &&
      request.auth.token.email == 'vivexatech@gmail.com';
    }

    function isAdminUser() {
      return request.auth != null &&
      (isSuperAdmin() || exists(/databases/$(database)/documents/users/$(request.auth.uid)));
    }

    function isStaffMember(staffId) {
      return request.auth != null &&
      staffId != null &&
      exists(/databases/$(database)/documents/staff/$(staffId)) &&
      (isSuperAdmin() || 
       isAdminUser() || 
       // Check if staff document email matches user's email
       // Note: Emails are normalized to lowercase in application code
       // Security rules have elevated privileges to read documents for verification
       (get(/databases/$(database)/documents/staff/$(staffId)).data.email == request.auth.token.email));
    }

    // Helper function to check if user is a staff member
    // Note: For list queries, we allow staff to query tasks
    // Document-level read rules will ensure they only see their own tasks
    function isOwnTask() {
      return request.auth != null &&
      resource.data.assigneeId != null &&
      isStaffMember(resource.data.assigneeId);
    }

    function isOwnAttendance() {
      return request.auth != null &&
      resource.data.staffId != null &&
      isStaffMember(resource.data.staffId);
    }

    // Users collection: Allow users to read their own document, super admin can read/write all
    match /users/{userId} {
      allow read: if request.auth != null && 
        (request.auth.uid == userId || isSuperAdmin());
      allow write: if isSuperAdmin();
    }

    match /portfolio/{docId} {
      allow read: if true;
      allow write: if isSuperAdmin();
    }

    match /intern-certificates/{docId} {
      allow read: if true;
      allow write: if isSuperAdmin();
    }

    match /clients/{docId} {
      allow read, write: if isSuperAdmin() || isAdminUser();
    }

    match /sales/{docId} {
      allow read, write: if isSuperAdmin() || isAdminUser();
    }

    match /invoices/{docId} {
      allow read, write: if isSuperAdmin() || isAdminUser();
    }

    match /staff/{docId} {
      allow read: if isSuperAdmin() || 
        isAdminUser() || 
        // Note: Emails are normalized to lowercase in application code
        (request.auth != null && resource.data.email == request.auth.token.email);
      // Allow list queries so staff can find their own record
      allow list: if request.auth != null;
      allow write: if isSuperAdmin() || isAdminUser();
    }

    match /tasks/{docId} {
      // Allow read for individual documents
      // Staff can only read tasks where assigneeId matches their staff record
      allow read: if isSuperAdmin() || 
        isAdminUser() || 
        (request.auth != null && 
         resource.data.assigneeId != null &&
         isStaffMember(resource.data.assigneeId));
      
      // Allow list queries for authenticated users
      // Document-level read rules above will filter results
      // This allows the query to proceed, then filters documents
      allow list: if request.auth != null;
      
      allow update: if isSuperAdmin() || 
        isAdminUser() || 
        (request.auth != null && 
         resource.data.assigneeId != null &&
         isStaffMember(resource.data.assigneeId) &&
         request.resource.data.assigneeId == resource.data.assigneeId &&
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status', 'progress', 'updatedAt']));
      allow create, delete: if isSuperAdmin() || isAdminUser();
    }

    match /attendance/{docId} {
      allow read: if isSuperAdmin() || 
        isAdminUser() || 
        (request.auth != null && 
         resource.data.staffId != null &&
         isStaffMember(resource.data.staffId));
      allow create: if isSuperAdmin() || 
        isAdminUser() || 
        (request.auth != null && 
         request.resource.data.staffId != null &&
         isStaffMember(request.resource.data.staffId) &&
         request.resource.data.staffId == request.resource.data.staffId);
      allow update: if isSuperAdmin() || 
        isAdminUser() || 
        (request.auth != null && 
         resource.data.staffId != null &&
         isStaffMember(resource.data.staffId) &&
         request.resource.data.staffId == resource.data.staffId);
      allow delete: if isSuperAdmin() || isAdminUser();
    }
  }
}
